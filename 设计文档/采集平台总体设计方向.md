### 4. 各适配器详细设计

#### 4.1 DB 适配器

| 项目 | 实现 |
|------|------|
| 连接池 | `sqlx::PgPool` / `MySqlPool` / `SqlitePool` |
| SQL 执行 | `sqlx::query(sql).fetch_all()` |
| 变量替换 | `${scan_start}`, `${scan_end}`, `${counter_key}` |
| 输出 | CSV: `|` 分隔, detail 文件描述字段 |
| 配置示例 | 见 4.6 |

```rust
impl DbAdapter {
    pub async fn collect(&self, task: CollectTask) -> Result<CollectResult> {
        // 1. 获取 SQL 模板 (来自 task.params["sql"])
        let sql_tpl = task.params.get("sql")
            .ok_or(anyhow!("DB adapter: missing 'sql' param"))?;

        // 2. 变量替换
        let sql = self.var_parser.replace(sql_tpl, &task.params);

        // 3. 执行查询
        let rows = sqlx::query(&sql)
            .fetch_all(&*self.pool)
            .await?;

        // 4. 列信息
        let columns: Vec<String> = rows.columns()
            .iter().map(|c| c.name().to_string())
            .collect();

        // 5. 写入 CSV
        let csv_path = output::csv_path(&task);
        let mut wtr = csv::WriterBuilder::new()
            .delimiter(b'|')
            .from_path(&csv_path)?;
        wtr.write_record(&columns)?;
        for row in &rows {
            let record: Vec<String> = columns.iter()
                .map(|c| row.get::<String, _>(c.as_str()).unwrap_or_default())
                .collect();
            wtr.write_record(&record)?;
        }
        wtr.flush()?;

        // 6. 写 detail
        output::write_detail(&task, &columns)?;

        Ok(CollectResult {
            record_count: rows.len() as u64,
            csv_path: Some(csv_path),
            ..
        })
    }
}
```

#### 4.2 FTP 适配器

| 阶段 | 实现 |
|------|------|
| 连接 | `suppaftp::Ftp::connect().login()` |
| 路径解析 | `${var}` 替换 + 分号 `;` 分隔多路径 |
| 文件搜索 | `list()` → 正则/内容/时间过滤 |
| 下载 | 支持 `max_download_threads` 并发 |
| 解压 | `.zip` 自动解压到临时目录 |
| 解析 | 正则逐行解析，输出 CSV |

文件搜索模式（继承自 Java 版）:

| 模式 | 行为 |
|------|------|
| `普通路径` | `list()` 后正则匹配文件名 |
| `content<` | 远程读取文件内容，逐行正则匹配 |
| `timeperiod<` | 按文件修改时间范围筛选 |
| `<filename:` | 精确指定文件名 |
| `cmd:` | SSH/Telnet 执行命令获取文件列表 |

#### 4.3 SFTP 适配器

```rust
pub struct SftpAdapter {
    session: Arc<SshSession>,
    remote_dir: String,
    file_pattern: Regex,
}

impl SftpAdapter {
    pub async fn collect(&self, task: CollectTask) -> Result<CollectResult> {
        let sftp = self.session.sftp().await?;

        // 1. 列出远程目录
        let entries = sftp.readdir(&self.remote_dir).await?;

        // 2. 按正则过滤
        let matched: Vec<_> = entries.iter()
            .filter(|e| e.0.is_file() && self.file_pattern.is_match(&e.1))
            .collect();

        // 3. 下载到临时目录
        let tmp = tempfile::tempdir()?;
        for (path, name) in &matched {
            let local = tmp.path().join(name);
            sftp.download(path, &local).await?;
        }

        // 4. 解析 → CSV
        parse_files_to_csv(tmp.path(), &task)
    }
}
```

#### 4.4 Kafka 适配器

```rust
pub struct KafkaAdapter {
    consumer: Arc<StreamConsumer>,
    producer: Arc<FutureProducer>,
}

impl KafkaAdapter {
    pub async fn collect(&self, task: CollectTask) -> Result<CollectResult> {
        let topic = task.params.get("topic").unwrap();
        let timeout_ms = task.params.get("timeout_ms")
            .and_then(|v| v.parse().ok()).unwrap_or(5000u64);
        let max_records = task.params.get("max_records")
            .and_then(|v| v.parse().ok()).unwrap_or(1000u64);

        // 1. 订阅 topic
        self.consumer.subscribe(&[topic])?;

        // 2. 拉取消息 (timeout 或 max_records 任一满足即停)
        let mut records = Vec::new();
        let timer = tokio::time::sleep(Duration::from_millis(timeout_ms));
        tokio::pin!(timer);

        loop {
            tokio::select! {
                msg = self.consumer.recv() => {
                    match msg {
                        Ok(m) => {
                            if let Some(payload) = m.payload() {
                                records.push(payload.to_vec());
                            }
                        }
                        Err(e) => { warn!("kafka error: {}", e); break; }
                    }
                    if records.len() >= max_records as usize { break; }
                }
                _ = &mut timer => break,
            }
        }

        // 3. 输出: raw/json → CSV
        write_records_to_csv(&task, &records)
    }
}
```

#### 4.5 RESTful 适配器

```rust
pub struct RestfulAdapter {
    client: reqwest::Client,
    var_parser: VarParser,
}

impl RestfulAdapter {
    pub async fn collect(&self, task: CollectTask) -> Result<CollectResult> {
        let method = task.params.get("method").map(String::as_str).unwrap_or("GET");
        let url = self.var_parser.replace(
            task.params.get("url").unwrap(), &task.params);

        // 1. 构建请求
        let mut req = match method {
            "GET"  => self.client.get(&url),
            "POST" => {
                let body = self.var_parser.replace(
                    task.params.get("body").unwrap_or(&String::new()), &task.params);
                self.client.post(&url).header("content-type", "application/json").body(body)
            }
            _ => bail!("unsupported HTTP method: {}", method),
        };
        // 添加自定义请求头
        for (k, v) in task.params.iter()
            .filter(|(k, _)| k.starts_with("header_")) {
            req = req.header(k.trim_start_matches("header_"), v);
        }
        // 2. 发送请求
        let resp = req.send().await?;
        let body = resp.text().await?;

        // 3. 解析响应 → CSV
        let records: Vec<Vec<String>> = match task.params
            .get("response_type").map(String::as_str) {
            Some("json") => {
                // JSON 路径提取: task.params["json_path"] = "$.data[*].key1,key2"
                let json: serde_json::Value = serde_json::from_str(&body)?;
                extract_json_to_rows(&json, task.params.get("json_path"))?
            }
            Some("xml") => {
                parse_xml_to_rows(&body, task.params.get("xpath"))?
            }
            _ => vec![vec![body]], // raw text
        };
        // 4. 写入 CSV
        write_csv_rows(&task, &records)
    }
}
```

### 6. 表达式变量引擎

```rust
// util/var_parser.rs

/// 变量替换引擎, 支持:
///   ${var_name}           — 简单变量
///   ${ENV_VAR}            — 环境变量
///   ${scan_start}         — 预定义时间变量
///   ${scan_start:yyyyMMdd}— 时间格式化
///   ${counter_key}        — 任务参数
pub struct VarParser {
    predefined: HashMap<String, Box<dyn Fn(&CollectTask) -> String + Send + Sync>>,
}

impl VarParser {
    pub fn new() -> Self {
        let mut p = HashMap::new();
        p.insert("scan_start".into(), Box::new(|t| t.scan_start.to_rfc3339()) as _);
        p.insert("scan_end".into(),   Box::new(|t| t.scan_end.to_rfc3339()) as _);
        // ...
        Self { predefined: p }
    }

    pub fn replace(&self, template: &str, task: &CollectTask) -> String {
        RE.replace_all(template, |caps: &Captures| {
            let name = &caps[1];
            // 1. 尝试预定义
            if let Some(f) = self.predefined.get(name) {
                return f(task);
            }
            // 2. 尝试任务参数
            if let Some(v) = task.params.get(name) {
                return v.clone();
            }
            // 3. 尝试系统环境变量
            if let Ok(v) = std::env::var(name) {
                return v;
            }
            caps[0].to_string() // 保持原样
        })
    }
}
```

### 4. 数据源配置表单设计 (动态表单)

根据不同数据源类型，动态渲染表单字段:

```typescript
// types/datasource.ts

type DataSourceType = 'db' | 'ftp' | 'sftp' | 'kafka' | 'restful';

interface DataSourceConfig {
  id: string;
  name: string;
  type: DataSourceType;
  enabled: boolean;
  description?: string;
  connection: ConnectionConfig;
  collect: CollectConfig;
  schedule?: ScheduleConfig;
  output?: OutputConfig;
  created_at: string;
  updated_at: string;
  status: 'online' | 'offline' | 'error';
  last_collect?: string;
}

// 按类型区分配置
type ConnectionConfig = DbConnection | FtpConnection | SftpConnection
                      | KafkaConnection | RestfulConnection;

interface DbConnection {
  url: string;
  max_connections: number;
  statement_timeout: number;
}

interface FtpConnection {
  host: string;
  port: number;
  user: string;
  password: string;       // 前端显示 ***, 接口传密文
  passive: boolean;
  encoding: string;
}

interface SftpConnection {
  host: string;
  port: number;
  user: string;
  auth_type: 'password' | 'key';
  password?: string;
  key_content?: string;   // 上传私钥文件
}

interface KafkaConnection {
  brokers: string;
  group_id: string;
  ssl?: boolean;
  sasl?: SaslConfig;
}

interface RestfulConnection {
  base_url: string;
  timeout_secs: number;
  tls_verify: boolean;
}
```

#### 4.1 新增数据源向导 (Element Plus Steps)

```vue
<!-- views/datasources/wizard/index.vue -->
<script setup lang="ts">
import { ref, computed } from 'vue'
import type { DataSourceType } from '@/types/datasource'

const currentStep = ref(0)
const formRef = ref()
const wizardType = ref<DataSourceType | null>(null)

const steps = [
  { title: '选择类型' },
  { title: '连接信息' },
  { title: '采集参数' },
  { title: '连接测试' },
  { title: '确认保存' },
]

const formData = ref({
  type: '',
  connection: {},
  collect: {},
  schedule: {},
})

function handleTypeSelect(type: DataSourceType) {
  wizardType.value = type
  formData.value.type = type
}

async function handleNext() {
  const valid = await formRef.value?.validate().catch(() => false)
  if (!valid) return
  currentStep.value++
}

function handlePrev() {
  if (currentStep.value > 0) currentStep.value--
}

async function handleSave() {
  // 调用 API 保存
}
</script>

<template>
  <el-card header="新增数据源">
    <el-steps :active="currentStep" finish-status="success" align-center>
      <el-step v-for="s in steps" :key="s.title" :title="s.title" />
    </el-steps>

    <div class="step-content">
      <StepSelectType v-if="currentStep === 0" @select="handleTypeSelect" />
      <StepConnection v-else-if="currentStep === 1" :type="wizardType" ref="formRef" v-model="formData.connection" />
      <StepCollection v-else-if="currentStep === 2" :type="wizardType" ref="formRef" v-model="formData.collect" />
      <StepTest v-else-if="currentStep === 3" :config="formData" />
      <StepConfirm v-else-if="currentStep === 4" :config="formData" />
    </div>

    <div class="step-actions">
      <el-button v-if="currentStep > 0" @click="handlePrev">上一步</el-button>
      <el-button v-if="currentStep < 4" type="primary" @click="handleNext">下一步</el-button>
      <el-button v-else type="primary" @click="handleSave">保存并启用</el-button>
    </div>
  </el-card>
</template>
```

```rust
// api/routes.rs

pub fn router(state: AppState) -> Router {
    Router::new()
        // 数据源管理
        .route("/api/v1/datasources",         get(list_datasources).post(create_datasource))
        .route("/api/v1/datasources/:id",     get(get_datasource).put(update_datasource).delete(delete_datasource))
        .route("/api/v1/datasources/:id/test",post(test_connection))
        .route("/api/v1/datasources/:id/enable",post(toggle_datasource))
        .route("/api/v1/datasources/:id/logs", get(get_datasource_logs))
        .route("/api/v1/datasources/types",    get(list_datasource_types))
        // 任务管理
        .route("/api/v1/tasks",               get(list_tasks).post(create_task))
        .route("/api/v1/tasks/:id",           get(get_task))
        .route("/api/v1/tasks/:id/trigger",   post(trigger_task))
        .route("/api/v1/tasks/:id/cancel",    post(cancel_task))
        // 监控
        .route("/api/v1/monitor/stats",       get(get_stats))
        .route("/api/v1/monitor/logs",        get(stream_logs))
        // 静态文件
        .nest_service("/", ServeDir::new("dist"))
        .with_state(state)
}
```

### 7. 数据模型 (Rust 侧持久化)

```rust
// 配置存储在 SQLite, 通过 Diesel 或 sqlx 操作

CREATE TABLE datasources (
    id          TEXT PRIMARY KEY,
    name        TEXT NOT NULL,
    type        TEXT NOT NULL,   -- 'db'|'ftp'|'sftp'|'kafka'|'restful'
    enabled     INTEGER NOT NULL DEFAULT 1,
    description TEXT,
    config_json TEXT NOT NULL,   -- 完整配置序列化为 JSON
    created_at  TEXT NOT NULL,   -- ISO8601
    updated_at  TEXT NOT NULL,
    status      TEXT DEFAULT 'offline',
    last_error  TEXT
);

CREATE TABLE collect_tasks (
    id              TEXT PRIMARY KEY,
    datasource_id   TEXT NOT NULL REFERENCES datasources(id),
    counter_name    TEXT NOT NULL,
    scan_start      TEXT NOT NULL,
    scan_end        TEXT NOT NULL,
    status          TEXT NOT NULL,  -- pending|running|success|failed
    record_count    INTEGER DEFAULT 0,
    csv_path        TEXT,
    error_msg       TEXT,
    created_at      TEXT NOT NULL,
    completed_at    TEXT
);

CREATE TABLE collect_logs (
    id              INTEGER PRIMARY KEY AUTOINCREMENT,
    task_id         TEXT REFERENCES collect_tasks(id),
    datasource_id   TEXT REFERENCES datasources(id),
    level           TEXT NOT NULL,  -- info|warn|error
    message         TEXT NOT NULL,
    created_at      TEXT NOT NULL
);
```
